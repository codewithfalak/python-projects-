import datetime

expenses = []

def add_expense():
    try:
        amount = float(input("Enter amount (₹): "))
        category = input("Enter category (Food, Travel, Shopping, etc.): ")
        date = datetime.date.today()
        expenses.append({"amount": amount, "category": category, "date": date})
        print("💰 Expense added!")
    except ValueError:
        print("❌ Invalid input!")

def show_expenses():
    if not expenses:
        print("👉 No expenses yet!")
    else:
        total = 0
        print("\n📊 Expenses:")
        for i, exp in enumerate(expenses, 1):
            print(f"{i}. ₹{exp['amount']} - {exp['category']} (on {exp['date']})")
            total += exp["amount"]
        print(f"\n💵 Total Spent: ₹{total}")

def filter_by_category():
    if not expenses:
        print("👉 No expenses yet!")
        return
    category = input("Enter category to filter: ")
    filtered = [exp for exp in expenses if exp["category"].lower() == category.lower()]
    if filtered:
        print(f"\n📂 Expenses in {category}:")
        for exp in filtered:
            print(f"- ₹{exp['amount']} (on {exp['date']})")
    else:
        print(f"❌ No expenses found in {category}")

while True:
    print("\nExpense Tracker Menu")
    print("1. Add Expense")
    print("2. Show All Expenses")
    print("3. Filter by Category")
    print("4. Exit")
    choice = input("Choose an option: ")

    if choice == "1":
        add_expense()
    elif choice == "2":
        show_expenses()
    elif choice == "3":
        filter_by_category()
    elif choice == "4":
        print("👋 Goodbye!")
        break
    else:
        print("❌ Invalid choice!")
